219 bool i2c_example_cargo_command(uint32_t command)
220 {
221  bool status = false;
222 
223  /* Make sure we do not have a race for i2c_application_command/i2c_application_status */
224  acc_integration_critical_section_enter();
225 
226  if (g_application_state.i2c_application_command == 0U)
227  {
228  /* Set Ready PIN to LOW while processing the command */
229  i2c_application_system_set_ready_pin(false);
230 
231  /* Set status BUSY bit */
232  g_application_state.i2c_application_status |= EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_BUSY_MASK;
233  g_application_state.i2c_application_command = command;
234  status = true;
235  }
236 
237  acc_integration_critical_section_exit();
238  return status;
239 }

320 int acconeer_main(int argc, char *argv[])
321 {
322  (void)argc;
323  (void)argv;
324 
325  bool setup_status = true;
326 
327  printf("I2C Example Cargo\n");
328  printf("Acconeer software version %s\n", acc_version_get());
329 
330  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
331 
332  if (acc_rss_hal_register(hal))
333  {
334  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_RSS_REGISTER_OK_MASK);
335  }
336  else
337  {
338  printf("ERROR: acc_rss_hal_register() failed\n\n");
339  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_APPLICATION_ERROR_MASK |
340  EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_RSS_REGISTER_OK_MASK);
341  setup_status = false;
342  }
343 
344  if (setup_status)
345  {
346  cargo_config_initialize(&g_resources.config, CARGO_PRESET_NO_LENS);
347  }
348 
349  /* Turn the sensor on */
350  acc_hal_integration_sensor_supply_on(SENSOR_ID);
351 
352  if (setup_status)
353  {
354  /* Create sensor */
355  create_sensor();
356  }
357 
358  i2c_application_system_init();
359 
360  /* Setup i2c register protocol */
361  example_cargo_reg_protocol_setup();
362 
363  while (true)
364  {
365 
366  /* Handle Commands */
367  uint32_t command = get_command();
368 
369  if (command == 0) // no command
370  {
371  if (i2c_application_system_test_wakeup_pin())
372  {
373  /* If wakeup is HIGH do WFI */
374  i2c_application_system_wait_for_interrupt();
375  }
376  else
377  {
378  /* If wakeup is LOW do SLEEP */
379 
380  /* Set ready pin LOW, we are about to power down */
381  i2c_application_system_set_ready_pin(false);
382 
383  uart_log("Enter low power state\n");
384  i2c_application_enter_low_power_state();
385 
386  /* We will restart here when WAKEUP goes HIGH */
387 
388  uart_log("Exit low power state\n");
389  }
390  i2c_application_system_set_ready_pin(true);
391 
392  /* No command to process */
393  continue;
394  }
395  /* Special command, always handle reset module command, even if error has occured */
396  if (command == EXAMPLE_CARGO_REG_COMMAND_ENUM_RESET_MODULE)
397  {
398  /* Reset system */
399  i2c_application_system_reset();
400  continue;
401  }
402 
403  if (application_status_test_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_APPLICATION_ERROR_MASK))
404  {
405  /* Do not process commands after error state */
406  continue;
407  }
408 
409  /* Handle command */
410  switch (command)
411  {
412  case EXAMPLE_CARGO_REG_COMMAND_ENUM_APPLY_CONFIGURATION:
413  if (!application_status_test_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_CONFIG_APPLY_OK_MASK))
414  {
415  // Apply configuration
416  apply_application_config();
417  }
418  break;
419  case EXAMPLE_CARGO_REG_COMMAND_ENUM_MEASURE_UTILIZATION_LEVEL:
420  acc_hal_integration_sensor_enable(SENSOR_ID);
421 
422  if (utilization_get_next())
423  {
424  log_cargo_result_on_uart(&g_resources.result);
425  }
426  else
427  {
428  printf("ERROR: Could not get next result\n");
429  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_APPLICATION_ERROR_MASK);
430  }
431 
432  acc_hal_integration_sensor_disable(SENSOR_ID);
433  break;
434  case EXAMPLE_CARGO_REG_COMMAND_ENUM_MEASURE_PRESENCE:
435  acc_hal_integration_sensor_enable(SENSOR_ID);
436 
437  if (!cargo_prepare_presence(g_resources.handle,
438  g_resources.sensor,
439  &g_resources.sensor_cal_result,
440  g_resources.buffer,
441  g_resources.buffer_size))
442  {
443  printf("ERROR: cargo_prepare_presence() failed\n");
444  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_APPLICATION_ERROR_MASK);
445  }
446  else
447  {
448  if (!enter_hibernation())
449  {
450  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_APPLICATION_ERROR_MASK);
451  }
452  else
453  {
454  uint32_t sleep_time_ms = (uint32_t)(1000.0f / g_resources.config.presence.update_rate);
455  i2c_application_set_periodic_wakeup(sleep_time_ms);
456 
457  presence_reset_burst_result();
458 
459  do
460  {
461  module_low_power();
462 
463  if (i2c_application_is_periodic_wakeup())
464  {
465  update_presence_update_rate_estimate();
466 
467  if (presence_get_next())
468  {
469  presence_update_burst_result(&g_resources.result);
470  log_cargo_result_on_uart(&g_resources.result);
471  }
472  else
473  {
474  printf("ERROR: Could not get next result\n");
475  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_APPLICATION_ERROR_MASK);
476  }
477  }
478 
479  } while (!application_status_test_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_APPLICATION_ERROR_MASK) &&
480  (cargo_current_mode_get(g_resources.handle) == CARGO_MODE_PRESENCE));
481 
482  exit_hibernation();
483  }
484  }
485 
486  i2c_application_set_periodic_wakeup(0U); // disable periodic wakeup
487 
488  acc_hal_integration_sensor_disable(SENSOR_ID);
489  break;
490  case EXAMPLE_CARGO_REG_COMMAND_ENUM_ENABLE_UART_LOGS:
491  g_application_state.uart_logs_enabed = true;
492  uart_log("UART logs enabled\n");
493  break;
494  case EXAMPLE_CARGO_REG_COMMAND_ENUM_DISABLE_UART_LOGS:
495  g_application_state.uart_logs_enabed = false;
496  uart_log("UART logs disabled\n");
497  break;
498  case EXAMPLE_CARGO_REG_COMMAND_ENUM_LOG_CONFIGURATION:
499  cargo_config_log(&g_resources.config);
500  break;
501  default:
502  printf("ERROR: Unknown command: %" PRIu32 "", command);
503  break;
504  }
505 
506  /* Command handling done, clear busy bit */
507  application_status_clr_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_BUSY_MASK);
508 
509  /* Set Ready PIN to HIGH when command processing is done */
510  i2c_application_system_set_ready_pin(true);
511  }
512 
513  return EXIT_FAILURE;
514 }

593 {
594  acc_hal_integration_sensor_enable(SENSOR_ID);
595 
596  application_status_clr_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_SENSOR_CALIBRATE_OK_MASK |
597  EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_SENSOR_CALIBRATE_ERROR_MASK);
598 
599  bool status;
600  bool cal_complete = false;
601 
602  do
603  {
604  status = acc_sensor_calibrate(g_resources.sensor, &cal_complete, &g_resources.sensor_cal_result, g_resources.buffer, g_resources.buffer_size);
605 
606  if (cal_complete)
607  {
608  break;
609  }
610 
611  if (status)
612  {
613  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
614  }
615  } while (status);
616 
617  if (status)
618  {
619  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_SENSOR_CALIBRATE_OK_MASK);
620  }
621  else
622  {
623  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_SENSOR_CALIBRATE_ERROR_MASK);
624  printf("ERROR: acc_sensor_calibrate() failed\n");
625  }
626 
627  /* Reset sensor after calibration by disabling it */
628  acc_hal_integration_sensor_disable(SENSOR_ID);
629 
630  return status;

633 static bool do_cargo_calibration(bool full_calibration)
634 {
635  bool status = true;
636  bool calibration_done = false;
637 
638  application_status_clr_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_CARGO_CALIBRATE_OK_MASK |
639  EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_CARGO_CALIBRATE_ERROR_MASK);
640 
641  acc_hal_integration_sensor_enable(SENSOR_ID);
642 
643  while (status && !calibration_done)
644  {
645  if (full_calibration)
646  {
647  status = cargo_calibrate(g_resources.handle,
648  g_resources.sensor,
649  &(g_resources.sensor_cal_result),
650  g_resources.buffer,
651  g_resources.buffer_size,
652  g_resources.distance_cal_result_static,
653  g_resources.distance_cal_result_static_size,
654  &(g_resources.distance_cal_result_dynamic),
655  &calibration_done);
656  }
657  else
658  {
659  status = cargo_update_calibration(g_resources.handle,
660  g_resources.sensor,
661  &(g_resources.sensor_cal_result),
662  g_resources.buffer,
663  g_resources.buffer_size,
664  &(g_resources.distance_cal_result_dynamic),
665  &calibration_done);
666  }
667 
668  if (status && !calibration_done)
669  {
670  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
671  }
672  }
673 
674  if (status)
675  {
676  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_CARGO_CALIBRATE_OK_MASK);
677  }
678  else
679  {
680  application_status_set_bits(EXAMPLE_CARGO_REG_APPLICATION_STATUS_FIELD_CARGO_CALIBRATE_ERROR_MASK);
681  }
682  return status;
683 }
