246 acc_vibration_handle_t *acc_vibration_handle_create(const acc_vibration_config_t *config)
247 {
248  uint16_t sweeps_per_frame = 0U;
249  float sweep_rate = 0.0f;
250 
251  acc_vibration_handle_t *handle = NULL;
252 
253  bool status = config != NULL;
254 
255  if (status)
256  {
257  status = validate_config(config);
258  }
259 
260  if (status)
261  {
262  handle = acc_integration_mem_alloc(sizeof(*handle));
263  status = handle != NULL;
264  }
265 
266  if (status)
267  {
268  handle->low_frequency_enhancement_enabled = config->low_frequency_enhancement;
269  sweeps_per_frame = config->sweeps_per_frame;
270  sweep_rate = config->sweep_rate;
271 
272  handle->has_init = false;
273 
274  setup_rfft_bounds(handle, config);
275 
276  handle->subframe_length = sweeps_per_frame;
277  handle->rfft_output_length = (handle->padded_time_series_length / 2U) + 1U;
278  handle->rfft_read_offset = 1U;
279  handle->data_length = handle->rfft_output_length - handle->rfft_read_offset;
280 
281  handle->continuous_data_acquisition = config->continuous_sweep_mode;
282 
283  handle->psd_to_radians_conversion_factor =
284  2.0f / (handle->continuous_data_acquisition ? (float)config->time_series_length : (float)sweeps_per_frame);
285 
286  handle->displacement_conversion_factor = handle->psd_to_radians_conversion_factor * RADIANS_TO_DISPLACEMENT_FACTOR;
287 
288  if (config->reported_displacement_mode == ACC_VIBRATION_REPORT_DISPLACEMENT_AS_PEAK2PEAK)
289  {
290  handle->displacement_conversion_factor *= 2.0f;
291  }
292  }
293 
294  if (status)
295  {
296  handle->sensor_config = acc_config_create();
297  status = handle->sensor_config != NULL;
298  }
299 
300  if (status)
301  {
302  status = translate_config(config, handle->sensor_config);
303  }
304 
305  if (status)
306  {
307  handle->time_series.buffer = acc_integration_mem_calloc(config->time_series_length, sizeof(*handle->time_series.buffer));
308  handle->time_series.write_idx = 0U;
309  handle->time_series.write_first_element = true;
310 
311  if (config->sweeps_per_frame < config->time_series_length && handle->continuous_data_acquisition == false)
312  {
313  handle->time_series.capacity = config->sweeps_per_frame;
314  }
315  else
316  {
317  handle->time_series.capacity = config->time_series_length;
318  }
319 
320  status = handle->time_series.buffer != NULL;
321  }
322 
323  if (status)
324  {
325  handle->frequencies = acc_integration_mem_calloc(handle->rfft_output_length, sizeof(*handle->frequencies));
326  status = handle->frequencies != NULL;
327  }
328 
329  if (status)
330  {
331  setup_sample_frequencies(handle, sweep_rate);
332  }
333 
334  if (status)
335  {
336  handle->double_buffer_filter_buffer = acc_integration_mem_calloc(sweeps_per_frame - 2U, sizeof(*handle->double_buffer_filter_buffer));
337  status = handle->double_buffer_filter_buffer != NULL;
338  }
339 
340  if (status)
341  {
342  handle->zero_mean_time_series = acc_integration_mem_calloc(config->time_series_length, sizeof(*handle->zero_mean_time_series));
343  status = handle->zero_mean_time_series != NULL;
344  }
345 
346  if (status)
347  {
348  handle->rfft_output = acc_integration_mem_calloc(handle->rfft_output_length, sizeof(*handle->rfft_output));
349  status = handle->rfft_output != NULL;
350  }
351 
352  if (status)
353  {
354  handle->lp_displacements = acc_integration_mem_calloc(handle->data_length, sizeof(*handle->lp_displacements));
355  status = handle->lp_displacements != NULL;
356  }
357 
358  if (status)
359  {
360  handle->threshold = acc_integration_mem_calloc(handle->data_length, sizeof(*handle->threshold));
361  status = handle->threshold != NULL;
362  }
363 
364  if (!status)
365  {
366  acc_vibration_handle_destroy(handle);
367  }
368 
369  return status ? handle : NULL;
370 }

459 void acc_vibration_process(acc_processing_result_t *proc_result,
460  acc_vibration_handle_t *handle,
461  acc_vibration_config_t *config,
462  acc_vibration_result_t *result)
463 {
464  const uint16_t sweeps_per_frame = config->sweeps_per_frame;
465 
466  result->max_sweep_amplitude = 0.0f;
467  result->time_series_std = FLT_MAX;
468 
469  result->peak_count = 0;
470  for (uint8_t i = 0; i < ACC_VIBRATION_MAX_REPORTED_PEAKS; i++)
471  {
472  result->peak_displacements[i] = 0.0f;
473  result->peak_frequencies[i] = 0.0f;
474  }
475 
476  /*
477  * Check if an object is close enough in front of the sensor to
478  * reach the configured threshold.
479  */
480 
481  for (uint16_t i = 0; i < handle->subframe_length; i++)
482  {
483  uint16_t frame_idx;
484 
485  if (handle->low_frequency_enhancement_enabled)
486  {
487  frame_idx = (2U * i) + 1U;
488  }
489  else
490  {
491  frame_idx = i;
492  }
493 
494  acc_int16_complex_t point = proc_result->frame[frame_idx];
495  result->max_sweep_amplitude = fmaxf(result->max_sweep_amplitude, cabsf(point.real + (point.imag * I)));
496  }
497 
498  if (result->max_sweep_amplitude < config->amplitude_threshold)
499  {
500  // Return without a (complete) result.
501  handle->has_init = false;
502  return;
503  }
504 
505  if (handle->low_frequency_enhancement_enabled)
506  {
507  /* Adjust measure frame with loopback frame */
508  for (uint16_t frame_idx = 0U; frame_idx < handle->subframe_length; frame_idx++)
509  {
510  uint16_t measure_src_idx = (2U * frame_idx) + 1U;
511  uint16_t loopback_src_idx = 2U * frame_idx;
512  uint16_t dst_idx = frame_idx;
513 
514  acc_int16_complex_t measure = proc_result->frame[measure_src_idx];
515  acc_int16_complex_t loopback = proc_result->frame[loopback_src_idx];
516 
517  complex float measure_cf = (float)measure.real + ((float)measure.imag * I);
518  complex float loopback_cf = (float)loopback.real + ((float)loopback.imag * I);
519 
520  complex float lb_norm = loopback_cf / cabsf(loopback_cf);
521  complex float compensated = measure_cf * conjf(lb_norm);
522 
523  proc_result->frame[dst_idx].real = (int16_t)(crealf(compensated) + 0.5f);
524  proc_result->frame[dst_idx].imag = (int16_t)(cimagf(compensated) + 0.5f);
525  }
526  }
527 
528  /*
529  * Process sensor data.
530  */
531 
532  // Handle frame based on whether or not continuous sweep mode is used.
533  if (handle->continuous_data_acquisition)
534  {
535  acc_algorithm_double_buffering_frame_filter(proc_result->frame, sweeps_per_frame, NUM_POINTS, handle->double_buffer_filter_buffer);
536  }
537  else
538  {
539  handle->time_series.write_first_element = true;
540  }
541 
542  for (uint16_t i = 0; i < handle->subframe_length; i++)
543  {
544  acc_int16_complex_t point = proc_result->frame[i];
545  float new_element = cargf(point.real + (point.imag * I));
546 
547  circular_float_buffer_write_angle(&handle->time_series, new_element);
548  }
549 
550  /*
551  * Estimate displacement per frequency.
552  */
553 
554  float *zero_mean_time_series = get_zero_mean_time_series(handle);
555 
556  acc_algorithm_rfft(zero_mean_time_series, config->time_series_length, handle->rfft_length_shift, handle->rfft_output);
557 
558  for (uint16_t i = 0; i < handle->data_length; i++)
559  {
560  float z_abs = cabsf(handle->rfft_output[i + handle->rfft_read_offset]);
561  float displacement = z_abs * handle->displacement_conversion_factor;
562 
563  if (handle->has_init)
564  {
565  handle->lp_displacements[i] = handle->lp_displacements[i] * config->lp_coeff + displacement * (1.0f - config->lp_coeff);
566  }
567  else
568  {
569  handle->lp_displacements[i] = displacement;
570  }
571  }
572 
573  handle->has_init = true;
574 
575  for (uint16_t i = 0U; i < config->time_series_length; i++)
576  {
577  zero_mean_time_series[i] *= RADIANS_TO_DISPLACEMENT_FACTOR;
578  }
579 
580  result->time_series_std = acc_algorithm_stddev_f32(zero_mean_time_series, config->time_series_length);
581 
582  update_vibration_result(handle, config, result);
583 }

589 static void circular_float_buffer_write_angle(circular_float_buffer_t *cb, float new_element)
590 {
591  if (cb->write_first_element == false)
592  {
593  float unwrap_buf[2];
594 
595  unwrap_buf[0] = circular_float_buffer_get_previous(cb);
596  unwrap_buf[1] = new_element;
597 
598  acc_algorithm_unwrap(unwrap_buf, 2U);
599 
600  cb->buffer[cb->write_idx] = unwrap_buf[1];
601  }
602  else
603  {
604  cb->buffer[cb->write_idx] = new_element;
605  cb->write_first_element = false;
606  }
607 
608  cb->write_idx = (cb->write_idx + 1U) % cb->capacity;
609 }
610 
611 static float circular_float_buffer_get(const circular_float_buffer_t *cb, uint16_t chronological_idx)
612 {
613  // Say a, b & c has just been written, then 'write_index' will point to the next index to
614  // write over (i.e. the oldest). We can use this to index the circular buffer in ascending
615  // chronological order (oldest to newest).
616  // | a | b | c | x | y | z |
617  // ^
618  uint16_t buffer_idx = (cb->write_idx + chronological_idx) % cb->capacity;
619 
620  return cb->buffer[buffer_idx];
621 }
622 
623 static float circular_float_buffer_get_previous(const circular_float_buffer_t *cb)
624 {
625  // circular_float_buffer_get wraps around to the previous value when called
626  // with cb->capacity - 1U as chronological_idx
627  float previous_element = circular_float_buffer_get(cb, cb->capacity - 1U);
628 
629  return previous_element;
630 }

632 static bool validate_config(const acc_vibration_config_t *config)
633 {
634  bool success = true;
635 
636  if (!IS_POWER_OF_TWO(config->time_series_length))
637  {

747 {

751  {

753  result->peak_displacements[insert_pos] = result->peak_displacements[insert_pos - 1];
754  result->peak_frequencies[insert_pos] = result->peak_frequencies[insert_pos - 1];

762 static void update_vibration_result(acc_vibration_handle_t *handle, acc_vibration_config_t *config, acc_vibration_result_t *result)
763 {
764  calculate_threshold(handle, config);
765 
766  uint32_t peak_count = 0;
767 
768  // Compare displacements to threshold (exclude first point as it does not form a peak)
769  for (uint16_t i = 1U; i < handle->data_length; i++)
770  {
771  if (handle->lp_displacements[i] > handle->threshold[i])
772  {
773  if (peak_count < ACC_VIBRATION_MAX_REPORTED_PEAKS)
774  {
775  uint32_t insert_pos = peak_count;
776  sort_displacements_desc(handle, i, insert_pos, result);
777  peak_count++;
778  }
779  else if (handle->lp_displacements[i] > result->peak_displacements[ACC_VIBRATION_MAX_REPORTED_PEAKS - 1U])
780  {
781  // We have reached the peak limit, insert and drop smallest.
782  uint32_t insert_pos = ACC_VIBRATION_MAX_REPORTED_PEAKS - 1U;
783  sort_displacements_desc(handle, i, insert_pos, result);
784  }
785  }
786  }
787 
788  result->peak_count = peak_count;
789 }

811 static void calculate_threshold(acc_vibration_handle_t *handle, acc_vibration_config_t *config)
812 {
813  for (uint16_t i = 0; i < handle->data_length; i++)
814  {
815  handle->threshold[i] = acc_algorithm_calculate_cfar(handle->lp_displacements,
816  handle->data_length,
817  CFAR_WINDOW_LENGTH,
818  CFAR_HALF_GUARD_LENGTH,
819  config->threshold_sensitivity,
820  i);
821  }
822 
823  /*
824  * Extend the CFAR threshold using extrapolation.
825  *
826  * The head is extended using linear extrapolation based on the first
827  * points of the original threshold.
828  *
829  * The tail is extended using the average of the last points of the
830  * original threshold.
831  */
832 
833  const float head_slope_multiplier = 2.0f;
834  const uint16_t head_slope_calculation_width = 2U;
835  const uint16_t tail_mean_calculation_width = 10U;
836  const uint16_t tail_mean_calculation_start = handle->data_length - CFAR_MARGIN - tail_mean_calculation_width;
837 
838  const float head_base_offset = handle->threshold[CFAR_MARGIN];
839  const uint16_t tail_offset = handle->data_length - CFAR_MARGIN;
840 
841  /*
842  * Calculate head.
843  */
844 
845  float head_mean_slope = 0.0f;
846 
847  for (uint16_t i = 0; i < head_slope_calculation_width; i++)
848  {
849  uint16_t pos = CFAR_MARGIN + i;
850  head_mean_slope += handle->threshold[pos + 1U] - handle->threshold[pos];
851  }
852 
853  head_mean_slope /= (float)head_slope_calculation_width;
854  head_mean_slope *= head_slope_multiplier;
855 
856  /*
857  * Calculate tail.
858  */
859 
860  float tail_mean = 0.0f;
861 
862  for (int16_t i = 0; i < tail_mean_calculation_width; i++)
863  {
864  tail_mean += handle->threshold[tail_mean_calculation_start + i];
865  }
866 
867  tail_mean /= (float)tail_mean_calculation_width;
868 
869  /*
870  * Extend the threshold at both ends.
871  */
872 
873  for (uint16_t i = 0; i < CFAR_MARGIN; i++)
874  {
875  handle->threshold[i] = ((float)i - CFAR_MARGIN) * head_mean_slope + head_base_offset;
876  handle->threshold[tail_offset + i] = tail_mean;
877  }
878 }
