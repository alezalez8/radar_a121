52 int acconeer_main(int argc, char *argv[])
53 {
54  (void)argc;
55  (void)argv;
56  acc_sensor_t *sensor = NULL;
57  acc_cal_result_t sensor_cal_result = {0};
58 
59  void *buffer = NULL;
60  uint32_t buffer_size = 0;
61 
62  uint32_t distance_cal_result_static_size = 0U;
63  uint8_t *distance_cal_result_static = NULL;
64  acc_detector_cal_result_dynamic_t distance_cal_result_dynamic = {0};
65 
66  cargo_handle_t *cargo_handle = NULL;
67  cargo_result_t cargo_result = {0};
68 
69  cargo_config_t cargo_config = {0};
70 
71  printf("Acconeer software version %s\n", acc_version_get());
72 
73  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
74 
75  if (!acc_rss_hal_register(hal))
76  {
77  return EXIT_FAILURE;
78  }
79 
80  cargo_config_initialize(&cargo_config, CARGO_PRESET_NO_LENS);
81 
82  cargo_config_log(&cargo_config);
83 
84  cargo_handle = cargo_handle_create(&cargo_config);
85  if (cargo_handle == NULL)
86  {
87  printf("cargo_handle_create() failed\n");
88  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
89  return EXIT_FAILURE;
90  }
91 
92  acc_detector_presence_metadata_t presence_metadata = {0};
93  if (cargo_presence_metadata_get(cargo_handle, &presence_metadata))
94  {
95  printf("Presence metadata\n");
96  printf("presence_metadata.start_m: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(presence_metadata.start_m));
97  printf("presence_metadata.end_m: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(presence_metadata.end_m));
98  printf("presence_metadata.step_length_m: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(presence_metadata.step_length_m));
99  printf("presence_metadata.num_points: %" PRIu16 "\n", presence_metadata.num_points);
100  }
101  else
102  {
103  printf("Cargo is not configured to have presence activated.\n");
104  }
105 
106  if (!cargo_get_buffer_size(cargo_handle, &buffer_size))
107  {
108  printf("cargo_get_buffer_size() failed\n");
109  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
110  return EXIT_FAILURE;
111  }
112 
113  if (!cargo_get_distance_cal_result_static_size(cargo_handle, &distance_cal_result_static_size))
114  {
115  printf("cargo_get_distance_cal_result_static_size() failed\n");
116  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
117  return EXIT_FAILURE;
118  }
119 
120  buffer = acc_integration_mem_alloc(buffer_size);
121  if (buffer == NULL)
122  {
123  printf("buffer allocation failed\n");
124  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
125  return EXIT_FAILURE;
126  }
127 
128  if (distance_cal_result_static_size > 0U)
129  {
130  distance_cal_result_static = acc_integration_mem_alloc(distance_cal_result_static_size);
131  if (distance_cal_result_static == NULL)
132  {
133  printf("distance_cal_result_static buffer allocation failed\n");
134  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
135  return EXIT_FAILURE;
136  }
137  }
138 
139  acc_hal_integration_sensor_supply_on(SENSOR_ID);
140  acc_hal_integration_sensor_enable(SENSOR_ID);
141 
142  sensor = acc_sensor_create(SENSOR_ID);
143  if (sensor == NULL)
144  {
145  printf("acc_sensor_create() failed\n");
146  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
147  return EXIT_FAILURE;
148  }
149 
150  if (!do_sensor_calibration(sensor, buffer, buffer_size, &sensor_cal_result))
151  {
152  printf("do_sensor_calibration() failed\n");
153  acc_sensor_status(sensor);
154  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
155  return EXIT_FAILURE;
156  }
157 
158  // Reset sensor after calibration by disabling/enabling it
159  acc_hal_integration_sensor_disable(SENSOR_ID);
160  acc_hal_integration_sensor_enable(SENSOR_ID);
161 
162  if (!do_cargo_calibration(sensor,
163  cargo_handle,
164  buffer,
165  buffer_size,
166  &sensor_cal_result,
167  distance_cal_result_static,
168  distance_cal_result_static_size,
169  &distance_cal_result_dynamic))
170  {
171  printf("do_cargo_calibration() failed\n");
172  acc_sensor_status(sensor);
173  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
174  return EXIT_FAILURE;
175  }
176 
177  while (true)
178  {
179  if (cargo_config.activate_utilization_level)
180  {
181  if (!cargo_prepare_utilization(cargo_handle, sensor, &sensor_cal_result, buffer, buffer_size))
182  {
183  printf("cargo_prepare_utilization() failed\n");
184  acc_sensor_status(sensor);
185  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
186  return EXIT_FAILURE;
187  }
188 
189  do
190  {
191  if (!acc_sensor_measure(sensor))
192  {
193  printf("acc_sensor_measure() failed\n");
194  acc_sensor_status(sensor);
195  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
196  return EXIT_FAILURE;
197  }
198 
199  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
200  {
201  printf("Sensor interrupt timeout\n");
202  acc_sensor_status(sensor);
203  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
204  return EXIT_FAILURE;
205  }
206 
207  if (!acc_sensor_read(sensor, buffer, buffer_size))
208  {
209  printf("acc_sensor_read() failed\n");
210  acc_sensor_status(sensor);
211  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
212  return EXIT_FAILURE;
213  }
214 
215  bool result_available = false;
216  if (!cargo_process_utilization(cargo_handle,
217  distance_cal_result_static,
218  &distance_cal_result_dynamic,
219  buffer,
220  &cargo_result,
221  &result_available,
222  NULL))
223  {
224  printf("cargo_process_utilization() failed\n");
225  acc_sensor_status(sensor);
226  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
227  return EXIT_FAILURE;
228  }
229 
230  print_cargo_result(&cargo_result, result_available);
231 
232  } while (cargo_current_mode_get(cargo_handle) == CARGO_MODE_UTILIZATION);
233  }
234  if (cargo_config.activate_presence)
235  {
236  if (!cargo_prepare_presence(cargo_handle, sensor, &sensor_cal_result, buffer, buffer_size))
237  {
238  printf("cargo_prepare_presence() failed\n");
239  acc_sensor_status(sensor);
240  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
241  return EXIT_FAILURE;
242  }
243 
244  do
245  {
246  if (!acc_sensor_measure(sensor))
247  {
248  printf("acc_sensor_measure() failed\n");
249  acc_sensor_status(sensor);
250  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
251  return EXIT_FAILURE;
252  }
253 
254  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
255  {
256  printf("Sensor interrupt timeout\n");
257  acc_sensor_status(sensor);
258  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
259  return EXIT_FAILURE;
260  }
261 
262  if (!acc_sensor_read(sensor, buffer, buffer_size))
263  {
264  printf("acc_sensor_read() failed\n");
265  acc_sensor_status(sensor);
266  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
267  return EXIT_FAILURE;
268  }
269 
270  if (!cargo_process_presence(cargo_handle, buffer, &cargo_result, NULL))
271  {
272  printf("cargo_process_presence() failed\n");
273  acc_sensor_status(sensor);
274  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
275  return EXIT_FAILURE;
276  }
277 
278  print_cargo_result(&cargo_result, true);
279 
280  } while (cargo_current_mode_get(cargo_handle) == CARGO_MODE_PRESENCE);
281  }
282  }
283 
284  cleanup(sensor, buffer, distance_cal_result_static, cargo_handle);
285 
286  printf("Application finished OK\n");
287 
288  return EXIT_SUCCESS;
289 }

291 static void cleanup(acc_sensor_t *sensor, void *buffer, uint8_t *distance_cal_result_static, cargo_handle_t *handle)
292 {
293  acc_hal_integration_sensor_disable(SENSOR_ID);
294  acc_hal_integration_sensor_supply_off(SENSOR_ID);
295 
296  if (sensor != NULL)
297  {
298  acc_sensor_destroy(sensor);
299  }
300 
301  if (buffer != NULL)
302  {
303  acc_integration_mem_free(buffer);
304  }
305 
306  if (distance_cal_result_static != NULL)
307  {
308  acc_integration_mem_free(distance_cal_result_static);
309  }
310 
311  if (handle != NULL)
312  {
313  cargo_handle_destroy(handle);
314  }
315 }

317 static bool do_sensor_calibration(acc_sensor_t *sensor, void *buffer, uint32_t buffer_size, acc_cal_result_t *cal_result)
318 {
319  bool status = false;
320  bool cal_complete = false;
321  const uint16_t calibration_retries = 1U;
322 
323  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
324  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
325  {
326  // Reset sensor before calibration by disabling/enabling it
327  acc_hal_integration_sensor_disable(SENSOR_ID);
328  acc_hal_integration_sensor_enable(SENSOR_ID);
329 
330  do
331  {
332  status = acc_sensor_calibrate(sensor, &cal_complete, cal_result, buffer, buffer_size);
333 
334  if (status && !cal_complete)
335  {
336  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
337  }
338  } while (status && !cal_complete);
339  }
340 
341  return status;
342 }

344 static bool do_cargo_calibration(acc_sensor_t *sensor,
345  cargo_handle_t *cargo_handle,
346  void *buffer,
347  uint32_t buffer_size,
348  acc_cal_result_t *sensor_cal_result,
349  uint8_t *distance_cal_result_static,
350  uint32_t distance_cal_result_static_size,
351  acc_detector_cal_result_dynamic_t *distance_cal_result_dynamic)
352 {
353  bool status = true;
354  bool calibration_done = false;
355 
356  while (status && !calibration_done)
357  {
358  status = cargo_calibrate(cargo_handle,
359  sensor,
360  sensor_cal_result,
361  buffer,
362  buffer_size,
363  distance_cal_result_static,
364  distance_cal_result_static_size,
365  distance_cal_result_dynamic,
366  &calibration_done);
367 
368  if (status && !calibration_done)
369  {
370  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
371  }
372  }
373 
374  return status;
375 }

377 static void print_cargo_result(const cargo_result_t *result, bool result_available)
378 {
379  static uint32_t frameno = 0U;
380  frameno++;
381 
382  printf("frameno %" PRIu32 "\n", frameno);
383 
384  if (result_available)
385  {
386  if (result->utilization_valid)
387  {
388  printf("distance: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(result->distance));
389  printf("level_m: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(result->level_m));
390  printf("level_percent: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(result->level_percent));
391  }
392 
393  if (result->presence_valid)
394  {
395  printf("presence_detected: %s\n", result->presence_detected ? "true" : "false");
396  printf("inter_presence_score: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(result->inter_presence_score));
397  printf("intra_presence_score: %" PRIfloat "\n", ACC_LOG_FLOAT_TO_INTEGER(result->intra_presence_score));
398  }
399  }
