39 void example_cargo_reg_read_protocol_status(uint32_t *value)
40 {
41  uint32_t error_flags = get_protocol_error_flags();
42  uint32_t protocol_status = 0;
43 
44  if ((error_flags & ACC_REG_ERROR_FLAG_PROTOCOL_STATE_ERROR) != 0)
45  {
46  protocol_status |= EXAMPLE_CARGO_REG_PROTOCOL_STATUS_FIELD_PROTOCOL_STATE_ERROR_MASK;
47  }
48 
49  if ((error_flags & ACC_REG_ERROR_FLAG_PACKET_LENGTH_ERROR) != 0)
50  {
51  protocol_status |= EXAMPLE_CARGO_REG_PROTOCOL_STATUS_FIELD_PACKET_LENGTH_ERROR_MASK;
52  }
53 
54  if ((error_flags & ACC_REG_ERROR_FLAG_ADDRESS_ERROR) != 0)
55  {
56  protocol_status |= EXAMPLE_CARGO_REG_PROTOCOL_STATUS_FIELD_ADDRESS_ERROR_MASK;
57  }
58 
59  if ((error_flags & ACC_REG_ERROR_FLAG_WRITE_FAILED) != 0)
60  {
61  protocol_status |= EXAMPLE_CARGO_REG_PROTOCOL_STATUS_FIELD_WRITE_FAILED_MASK;
62  }
63 
64  if ((error_flags & ACC_REG_ERROR_FLAG_WRITE_TO_READ_ONLY) != 0)
65  {
66  protocol_status |= EXAMPLE_CARGO_REG_PROTOCOL_STATUS_FIELD_WRITE_TO_READ_ONLY_MASK;
67  }
68 
69  *value = protocol_status;
70 }

281 void example_cargo_reg_read_result_header(uint32_t *value)
282 {
283  cargo_result_t result = {0};
284  int16_t temperature = i2c_example_cargo_get_temperature();
285 
286  uint32_t temperature_bits =
287  (((uint32_t)temperature) << EXAMPLE_CARGO_REG_RESULT_HEADER_FIELD_TEMPERATURE_POS) & EXAMPLE_CARGO_REG_RESULT_HEADER_FIELD_TEMPERATURE_MASK;
288 
289  i2c_example_cargo_copy_utilization_result_fields(&result);
290 
291  uint32_t utilization_valid_bit = (((uint32_t)result.utilization_valid) << EXAMPLE_CARGO_REG_RESULT_HEADER_FIELD_UTILIZATION_LEVEL_VALID_POS) &
292  EXAMPLE_CARGO_REG_RESULT_HEADER_FIELD_UTILIZATION_LEVEL_VALID_MASK;
293 
294  i2c_example_cargo_copy_presence_burst_result_fields(&result);
295 
296  uint32_t presence_valid_bit = (((uint32_t)result.presence_valid) << EXAMPLE_CARGO_REG_RESULT_HEADER_FIELD_PRESENCE_VALID_POS) &
297  EXAMPLE_CARGO_REG_RESULT_HEADER_FIELD_PRESENCE_VALID_MASK;
298 
299  uint32_t packed = temperature_bits | utilization_valid_bit | presence_valid_bit;
300  *value = packed;
301 }
